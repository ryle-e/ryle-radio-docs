117 // stores a new RadioTrackPlayer and alerts any observers
118 protected void PlayerCreation(RadioTrackPlayer _player)
119 {
120 foreach (RadioObserver observer in Observers)
121 {
122 // if an observer is watching this track
123 if (observer.AffectedTracks.Contains(_player.TrackW.name))
124 {
125 observer.AssignEvents(_player); // point it towards this new player
126 }
127 }
128
129 // below here is a section where we insert the new player into the list while preserving the order of the tracks in the Data inspector.
130 // we do this so that attenuation is preserved. if we just added players to the list one by one, the order would change, and so tracks
131 // that are supposed to get quieter when another is played (attenuate) will not do so if the other player was created afterwards.
132
133 // the index of the track in the RadioData track list- the order of the Data in the inspector
134 int indexInData = Data.TrackIDs.IndexOf(_player.TrackW.id);
135
136 // the index at which to put the newly created player
137 int indexForNewPlayer = players.Count;
138
139 // search through the currently existent players to find one with a higher index in Data
140 for (int i = 0; i < players.Count; i++)
141 {
142 if (Data.TrackIDs.IndexOf(players[i].TrackW.id) > indexInData)
143 {
144 indexForNewPlayer = i;
145 break;
146 }
147 }
148
149 // store the player
150 if (players.Count == 0)
151 players.Add(_player);
152 else
153 players.Insert(indexForNewPlayer, _player);
154 }

174 // play a track as a one-shot- a one-shot destroys itself after the track ends
175 public RadioTrackPlayer PlayOneShot(string _id)
176 {
177 // get the track with the given id
178 if (Data.TryGetTrack(_id, out RadioTrackWrapper track))
179 {
180 // create a new player for the track, set to one-shot
181 RadioTrackPlayer player = new(track, RadioTrackPlayer.PlayerType.OneShot, baseSampleRate);
182
183 // store the player
184 lock (playEvents)
185 playEvents += () => PlayerCreation(player);
186
187 // ensure that the new player is cleaned up when it's destroyed
188 player.DoDestroy += player =>
189 {
190 lock (playEvents)
191 playEvents += () => players.Remove(player);
192 };
193
194 // return the player
195 return player;
196 }
197 // if it can't find a player with the id, warn the user
198 else
199 {
200 Debug.LogWarning($"Can't find track with id {_id} to play as a one-shot!");
201 return null;
202 }

230 // try to find an active RadioTrackPlayer on this output
231 public bool TryGetPlayer(
232 string _trackID,
233 out RadioTrackPlayer _player,
234 bool _createNew,
235 MultiplePlayersSelector _multiplePlayerSelector = MultiplePlayersSelector.Youngest
236 )
237 {
238 // find any players associated with the track with the given id
239 var found = players.Where(p => p.TrackW.id == _trackID);
240
241 // if there are multiple players,
242 if (found.Count() > 1)
243 {
244 // select one based off the given selector
245 switch (_multiplePlayerSelector)
246 {
247 default: // choose the youngest player
248 case MultiplePlayersSelector.Youngest:
249 _player = found.Last();
250 break;
251
252 // choose the oldest player
253 case MultiplePlayersSelector.Oldest:
254 _player = found.First();
255 break;
256
257 // choose a random player
258 case MultiplePlayersSelector.Random:
259 int index = Random.Range(0, found.Count());
260 _player = found.ElementAt(index);
261
262 break;
263 }
264
265 // a player was found, so return true
266 return true;
267 }
268 // if there is one player,
269 else if (found.Count() > 0)
270 {
271 // choose it and return true
272 _player = found.First();
273 return true;
274 }
275
276 // if there aren't any players for this track,
277
278 // and the method is set to make a new one,
279 else if (_createNew)
280 {
281 // create a new player for this track
282 // if you want it to be a loop, play it manually- this is more of an error catch than an actual method of creation
283 _player = PlayOneShot(_trackID);
284 return true; // and return true
285 }
286 // and we aren't creating a new one,
287 else
288 {
289 _player = null; // there is no player
290 return false; // so we return false
291 }
292 }

294 // the core method- this is where the audio is output to the linked AudioSource
295 // this method appears to have initially been introduced for custom audio filters, but we can use it for custom audio output too
296 //
297 // _data here is whatever other audio is playing from the source- normally is nothing, unless you wanted some other audio playing
298 // if you do want this i would recommend using a separate audiosource so that you can control the volume separately
299 protected virtual void OnAudioFilterRead(float[] _data, int _channels)
300 {
301 // the output only plays back one channel, so we have to account for this when the radio is using
302 // tracks with more than one channel
303
304 // the number of samples for, total- the _data array has an entry for each channel by default. e.g if _data was 2048
305 // entries long, and there were two channels- there would actually only be 1024 samples
306 int monoSampleCount = _data.Length / _channels;
307
308 // for every sample in the data array
309 for (int index = 0; index < monoSampleCount; index++)
310 {
311 // we want to store the sample itself, and the added volume of all previous tracks
312 float sample = 0;
313 float combinedVolume = 0;
314
315 // for every active track,
316 foreach (RadioTrackPlayer player in players)
317 {
318 // get the audio in this sample
319 sample += player.GetSample(Tune, cachedPos, combinedVolume, out float outVolume, true);
320
321 // then move along to the next sample
322 player.IncrementSample();
323
324 // store the volume so far so that tracks with attenuation can adjust accordingly- see RadioTrackPlayer.GetSample()
325 combinedVolume += outVolume;
326 }
327
328 // this function uses _data for each sample packed into one big list- each channel is joined end to end
329 // that means if there are multiple _channels, we need to apply the audio to each of them before jumping to the next sample
330 // therefore we iterate through every channel, for every sample
331 int indexWithChannels = index * _channels;
332
333 // for each channel,
334 for (int channel = indexWithChannels; channel < indexWithChannels + _channels; channel++)
335 _data[channel] += sample; // apply the sample
336 }
337
338 // execute all events waiting for this read sequence to end
339 lock (playEvents)
340 {
341 playEvents(); // execute the delegate
342 playEvents = () => { }; // clear it
343 }
344 }
