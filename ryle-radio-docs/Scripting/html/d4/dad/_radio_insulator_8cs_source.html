9 // an insulator for a RadioTrack- the further in the output, the quieter the track
10 // this has a custom editor in RadioInsulatorEditor.cs
11 [AddComponentMenu("Ryle Radio/Radio Insulator")]
12 public class RadioInsulator : RadioComponentDataAccessor
13 {
14 // the inner and outer sizes of the insulator- if it's in the inner box, insulation power is at the maximum. if it's between the outer and inner
15 // boxes, insulation power is between the minimum and the maximum. if it's outside both, insulation power is at the minimum.
16 [Space(8)]
17 public Vector3 innerBoxSize = Vector3.one * 0.9f;
18 public Vector3 outerBoxSize = Vector3.one;
19
20 [Space(8)]
21
22 // the minimum and maximum insulation- how high the insulation is when the output is inside the inner box or between the inner and
23 // outer boxes
24 [SerializeField, MinMaxSlider(0, 1)]
25 private Vector2 insulation = new(0, 0.5f);
26
27 // the curve that defines insulation between the inner and outer boxes
28 [SerializeField, CurveRange(0, 0, 1, 1)]
29 private AnimationCurve insulationCurve = AnimationCurve.Linear(0, 0, 1, 1);
30
31 // if minimum insulation is greater than 0, we need to know if it applies to all outputss outside of the insulator as well
32 // if this is set to true and min insulation is set to 0.1, for example, every output outside of the insulator will still have 0.1 insulation
33 // applied to it
34 [SerializeField, AllowNesting, ShowIf("ShowApplyToAllOutputs")]
35 private bool applyToAllOutputsOutside = false;
36
37 // the position of this insulator at the last frame
38 // we cache this as we cannot use transform.position in GetPower, as audio is on a different thread and we would otherwise get errors
39 private Vector3 cachedPos;
40
41 // the inner and outer boxes with transform.localScale applied
42 public Vector3 InnerBoxSizeAdjusted { get; private set; }
43 public Vector3 OuterBoxSizeAdjusted { get; private set; }
44
45 private bool ShowApplyToAllOutputs => insulation.x > 0;
46
47

49 {

52 OuterBoxSizeAdjusted = transform.localToWorldMatrix * outerBoxSize;

54 // caches position for this frame
55 cachedPos = transform.position;
56 }

71 // gets the power of this insulator with the output at position
72 public float GetPower(Vector3 _position)
73 {
74 float power = 0;
75
76 // converts the boxes to bounds so that we can use the basic methods in the bounds class
77 Bounds innerBounds = new Bounds(cachedPos, InnerBoxSizeAdjusted);
78 Bounds outerBounds = new Bounds(cachedPos, OuterBoxSizeAdjusted);
79
80 // if the output is in the inner bounds, insulation is at maximum
81 if (innerBounds.Contains(_position))
82 power = 1;
83
84 // if the output is outside the outer bounds, insulation is at minimum
85 else if (!outerBounds.Contains(_position))
86 power = 0;
87
88 // if the output is between the inner and outer bounds, calculate how far between them it is
89 else
90 {
91 // originally i used a signed distance field for this, but for reasons i do not recall it didn't end up working as i wanted-
92 // i eventually worked out a way to calculate how far between a and b the point is using rays and bounds
93
94 // get the direction from the center of the boxes towards the given position
95 Vector3 dir = (_position - outerBounds.center).normalized;
96
97 // scale that distance by the size of the larger box so that it sits just outside (or on the corner of) the largest box
98 Vector3 scaledDir = new Vector3(dir.x * outerBounds.size.x, dir.y * outerBounds.size.y, dir.z * outerBounds.size.z);
99
100 // create a ray from the center of the box to the point just outside
101 Ray ray = new Ray(outerBounds.center, scaledDir);
102
103 // use this ray to find the distance from the center of the closest point on the outside of the inner and outer boxes
104 innerBounds.IntersectRay(ray, out float distance2);
105 outerBounds.IntersectRay(ray, out float distance);
106
107 // find the closest points on the faces of the boxes using this distance
108 Vector3 closestOnInner = innerBounds.center + ray.direction.normalized * -distance2;
109 Vector3 closestOnOuter = outerBounds.center + ray.direction.normalized * -distance;
110
111 // get the distance between the outer and inner points
112 float outerInnerDistance = Vector3.Distance(closestOnOuter, closestOnInner);
113
114 // get the distance from the inner to the given position
115 float d = Vector3.Distance(_position, closestOnInner);
116
117 // work out how far between the inner and outer points the given position is
118 power = 1 - d / outerInnerDistance;
119 }
120
121 // adjust the power by the insulation curve
122 float adjustedT = insulationCurve.Evaluate(power);
123
124 // if it's outside the boxes and does not universally apply, set the insulation to 0
125 if (adjustedT <= 0 && (ShowApplyToAllOutputs && !applyToAllOutputsOutside))
126 return 0;
127
128 // return the actual insulation using the min and max values
129 return Mathf.Lerp(insulation.x, insulation.y, adjustedT);
130 }
131 }
