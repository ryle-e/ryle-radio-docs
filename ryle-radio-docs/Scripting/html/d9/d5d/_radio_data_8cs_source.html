76 // fill the track id and id lists to match the current tracks
77 private void PopulateTrackIDs()
78 {
79 // if there are no tracks, don't try to get the names
80 if (trackWs.Count <= 0)
81 return;
82
83 // reset the track info lists
84 trackNames = new List<string>();
85 trackIDs = new List<string>();
86
87 // for every track in the radio
88 for (int i = 0; i < TrackWrappers.Count; i++)
89 {
90 // cache the track
91 RadioTrackWrapper track = TrackWrappers[i];
92
93 // find any other tracks with the same id as this one
94 var othersWithID = trackIDs.Where(t => t == track.id);
95
96 // if there are others, change the id of this track to match
97 if (othersWithID.Count() > 0)
98 track.id += othersWithID.Count(); // you can only add one new track at a time, so we just append the count to the end
99 // e.g a adding a track with the last one's id "music" will make the id of the new track "music1"
100
101 // combine some track info into a display id
102 string name = $"{track.id}, {track.range.x} - {track.range.y}";
103
104 // store the id and id
105 trackNames.Add(name);
106 trackIDs.Add(track.id);
107
108 // assign the id
109 track.name = name;
110 }
111 }

140 // attempt to get a track from this radio using an id or id
141 public bool TryGetTrack(string _idOrName, out RadioTrackWrapper _trackW, bool _useID = true)
142 {
143 // either an id or id can be supplied here, but we always convert it to an id for this
144 string id = "";
145
146 if (_useID) // if an id was provided, use it
147 id = _idOrName;
148 else // if a id was provided, convert it to the id
149 id = NameToID(_idOrName);
150
151 // find any track with that id
152 var found = TrackWrappers.Find(t => t.id == id);
153
154 // if a track is found,
155 if (found != null)
156 {
157 // output it and return true
158 _trackW = found;
159 return true;
160 }
161 else // otherwise
162 {
163 // output it and return false
164 _trackW = null;
165 return false;
166 }
