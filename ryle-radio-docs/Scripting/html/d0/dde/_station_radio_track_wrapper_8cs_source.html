107 // gets a new track of type depending on the provided id
108 // note that this is static
109 // also note that this uses IStationTrack, unlike RadioTrackWrapper
110 public static IStationTrack CreateTrackEditor(string _name)
111 {
112 // get the index of the chosen track type
113 int index = Array.IndexOf(TrackNames, _name);
114
115 // if you somehow have an invalid track type, don't create anything
116 if (index < 0)
117 return null;
118
119 // create the track generically
120 // more info in RadioTrackWrapper.CreateTrackEditor
121 IStationTrack outTrack = (IStationTrack)Activator.CreateInstance(TrackTypes[index]);
122 outTrack.IsInStation = true;
123
124 // return the track
125 return outTrack;
126 }

134 // get a sample from this track
135 public float GetSample(int _sampleIndex)
136 {
137 // if the track has a rest at the beginning at that sample index, return a silent sample
138 if (_sampleIndex < startAndEndRests.x)
139 return 0;
140 // same thing but for a rest at the end of the track
141 else if (_sampleIndex > (track.SampleCount + startAndEndRests.x))
142 return 0;
143
144 // otherwise, return the sample from the track
145 return track.GetSample(_sampleIndex);
146 }
