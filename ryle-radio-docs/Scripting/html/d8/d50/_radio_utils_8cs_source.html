38 // gets all types derived from a given type
39 public static Type[] FindDerivedTypes(Type baseType)
40 {
41 // we use all assemblies here for ease of use to developers creating and testing new tracks
42
43 // get all assemblies used in the project
44 Assembly[] assemblies = AppDomain.CurrentDomain.GetAssemblies();
45 IEnumerable<Type> types = new List<Type>();
46
47 // search through every assembly applicable classes
48 foreach (Assembly assembly in assemblies)
49 types = types.Union(assembly.GetTypes().Where(t => // find classes that,
50 t != baseType // aren't the given type
51 && baseType.IsAssignableFrom(t) // inherit from the given type
52 && !t.IsInterface // aren't interfaces
53 && !t.IsGenericType // aren't generic
54 && !t.IsAbstract // aren't abstract
55 ));
56
57 // returns the types
58 return types.ToArray();
59 }
