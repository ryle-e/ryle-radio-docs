9 // a broadcaster for a RadioTrack- the closer the output is, the louder the track
10 // this has a custom inspector in RadioBroadcasterEditor.cs
11 [AddComponentMenu("Ryle Radio/Radio Broadcaster")]
12 public class RadioBroadcaster : RadioComponentDataAccessor
13 {
14 // the inner and outer radii of this broadcaster- if the output is in the inner radius, the broadcast radiusProg is 1. if it's between the inner
15 // and outer radii, the radiusProg is between 0 and 1. if it's outside both, the radiusProg is 0.
16 [Space(8)]
17 public Vector2 broadcastRadius;
18
19 // the min and max broadcast power of this broadcaster- the first value is the power right on the outer edge of the broadcaster, and the second is the
20 // power in the inner radius. if the outer power is > 0, you can make it global in applyToAllOutputsOutside
21 [Space(8), MinMaxSlider(0, 1), SerializeField]
22 protected Vector2 broadcastPowers = new(0, 1);
23
24 // the falloff between the inner and outer broadcast ranges- you probably don't need to touch this but it's here if needed
25 [SerializeField, CurveRange(0, 0, 1, 1)]
26 protected AnimationCurve distanceFalloff = new(new Keyframe[2] {
27 new(0, 1, 0, 0),
28 new(1, 0, 0, 0)
29 });
30
31 // if the min broadcast power is > 0, it means that the broacast doesn't entirely stop once it hits the edge- this means we could make it
32 // global when outside the broadcast radius.
33 [SerializeField, AllowNesting, ShowIf("ShowApplyToAllOutputs")]
34 private bool applyToAllOutputsOutside = false;
35
36 // the position of the broadcaster at the last frame
37 // we cache this as we cannot use transform.position in GetPower, as audio is on a different thread and we would otherwise get errors
38 private Vector3 cachedPos;
39
40 private bool ShowApplyToAllOutputs => broadcastPowers.x > 0;
41
42
43 private void Update()
44 {
45 // cache the position
46 cachedPos = transform.position;
47 }
48
49 // link and unlink this broadcaster to a track
50 protected override void AssignToTrack(RadioTrackWrapper _track)
51 {
52 _track.broadcasters.Add(this);
53 _track.OnAddBroadcaster(this, _track);
54 }
55
56 protected override void RemoveFromTrack(RadioTrackWrapper _track)
57 {
58 _track.broadcasters.Remove(this);
59 _track.OnRemoveBroadcaster(this, _track);
60 }
61
62 // get the radiusProg of the output at a certain position relative to this broadcaster
63 // i.e a value from 0-1 for how close the output is to this broadcaster, that can be used as a multiplier to the audio
64 public float GetPower(Vector3 _receiverPos)
65 {
66 // the distance between the output and this broadcaster
67 float distance = Vector3.Distance(cachedPos, _receiverPos);
68
69 // how far between the inner and outer radii the distance is
70 float radiusProg = Mathf.Clamp01(Mathf.InverseLerp(broadcastRadius.x, broadcastRadius.y, distance));
71
72 // get the adjusted power according to the distance falloff
73 float adjustedT = distanceFalloff.Evaluate(radiusProg);
74
75 // if the output is outside of the range of this broadcaster and it's not global when outside, return a power of 0
76 if (adjustedT <= 0 && (ShowApplyToAllOutputs && !applyToAllOutputsOutside))
77 return 0;
78
79 // return the actual broadcast power according to values supplied in the broadcastPowers variable
80 return Mathf.Lerp(broadcastPowers.x, broadcastPowers.y, adjustedT);
81 }
82 }
