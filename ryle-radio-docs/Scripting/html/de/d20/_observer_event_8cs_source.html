63 // if this event needs a value to compare to
64 public bool NotNeedComparison => !NeedComparison;
65 public bool NeedComparison =>
66 type == EventType.OutputVolume
67 || type == EventType.Gain
68 || type == EventType.TunePower
69 || type == EventType.BroadcastPower
70 || type == EventType.Insulation
71 || type == EventType.OutputTune;
72
73 // if this event needs a range to compare to
74 public bool NotNeedVector => !NeedVector;
75 public bool NeedVector =>
76 comparison == ComparisonType.BetweenInclusive
77 || comparison == ComparisonType.BetweenExclusive;
78
79 // if this event is set to track OutputTune
80 public bool NotIsTune => !IsTune;
81 public bool IsTune =>
82 type == EventType.OutputTune;
83
84
85 // if this event is a comparison, check if the comparison is satisfied
86 public bool EvaluateComparison(float _cValue)
87 {
88 // if the event is not a comparison you should not be calling this method (unless it's being used on many events at once)
89 if (NotNeedComparison)
90 return false;
91
92 float value;
93 Vector2 range;
94
95 // if this event uses OutputTune, then make sure we're comparing to the tune values
96 if (IsTune)
97 {
98 range = tuneRange;
99 value = tuneValue;
100 }
101 else
102 {
103 value = clampedValue;
104 range = clampedRange;
105 }
106
107 // evaluate the specific comparison- just logical operators
108 return comparison switch
109 {
110 ComparisonType.Equal => _cValue == value,
111 ComparisonType.GreaterThan => _cValue > value,
112 ComparisonType.GreaterThanOrEqual => _cValue >= value,
113 ComparisonType.LessThan => _cValue < value,
114 ComparisonType.LessThanOrEqual => _cValue <= value,
115
116 ComparisonType.BetweenExclusive => range.x < _cValue && _cValue < range.y, // i want (x < value < y) notation >:)
117 ComparisonType.BetweenInclusive => range.x <= _cValue && _cValue <= range.y,
118
119 _ => false,
120 };
121 }
122
123 public override string ToString()
124 {
125 return type + "/" + comparison;
126 }
