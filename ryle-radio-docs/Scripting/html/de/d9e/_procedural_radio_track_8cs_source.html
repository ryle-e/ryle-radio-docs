15 WhiteNoise,
16 PinkNoise,
17 BrownNoise,

22 private const float NOISE_MULTIPLIER = .2f; // the noise is much louder by default due to the random being between -1 and 1, so we shrink it
23 private const float PINK_MULTIPLIER = .5f; // pink noise is louder so we curb it a little
24 private const float BASE_SAMPLE_RATE = 44100; // the default sample rate for the procedural tracks, can adjust this if needed
25
26 public ProceduralType proceduralType = ProceduralType.WhiteNoise;
27
28 // if this is in a station, then the audio needs to end sometime- give it a duration
29 // if this is not in a station then the audio is endless anyway
30 [AllowNesting, ShowIf("IsInStation")]
31 public float duration = 0;
32
33 // frequency (pitch) of the generated wave
34 [AllowNesting, ShowIf("proceduralType", ProceduralType.SineWave), Range(1, 2000)]
35 public float waveFrequency = 100;
36
37 // brown noise works by adding the generated number to all previous ones
38 // this float is what previous ones are multiplied by when they're stored
39 // therefore the higher the walk power, the closer it is to white noise
40 [AllowNesting, ShowIf("proceduralType", ProceduralType.BrownNoise), Range(0, 1)]
41 public float brownWalkPower = 0.5f;
42
43 // we can't use UnityEngine.Random during audio updates as it's on a different thread, so we need to use System.Random
44 private System.Random random;
45
46 // the progress through the waveform when using wave audio
47 private float phase = 0;

67 public override float GetSample(int _sampleIndex)
68 {
69 // all the noise algorithms use a white noise float
70 float white = 0;
71
72 switch (proceduralType)
73 {
74 // generate a random value between -1 and 1
75 case ProceduralType.WhiteNoise:
76 white = ((float)random.NextDouble() * 2) - 1;
77 return white * NOISE_MULTIPLIER;
78
79 // generate a random value with a pink noise filter applied
80 case ProceduralType.PinkNoise:
81 white = ((float)random.NextDouble() * 2) - 1;
82
83 // generated using paul kellet's refined method (pk3)
84 // https://www.firstpr.com.au/dsp/pink-noise/#Filtering:~:text=(This%20is%20pke,p2%20%2B%20white%20*%200.1848%3B
85 p0 = 0.99886f * p0 + white * 0.0555179f;
86 p1 = 0.99332f * p1 + white * 0.0750759f;
87 p2 = 0.96900f * p2 + white * 0.1538520f;
88 p3 = 0.86650f * p3 + white * 0.3104856f;
89 p4 = 0.55000f * p4 + white * 0.5329522f;
90 p5 = -0.7616f * p5 - white * 0.0168980f;
91 float pink = p0 + p1 + p2 + p3 + p4 + p5 + p6 + white * 0.5362f;
92 p6 = white * 0.115926f;
93
94 // pink is louder than normal noise so we use a secondary multiplier on it
95 return pink * NOISE_MULTIPLIER * PINK_MULTIPLIER;
96
97 // generate a random value, using previous values to soften the noise
98 // theory explained by gemini, adjusted from https://forum.juce.com/t/creating-colored-noise/30012/4
99 case ProceduralType.BrownNoise:
100 white = ((float)random.NextDouble() * 2) - 1;
101
102 lastBrown += white * brownWalkPower;
103 lastBrown = Mathf.Clamp(lastBrown, -1, 1) * 0.998f; // use a small <1 constant to ensure the noise doesn't get constantly louder
104
105 return lastBrown * NOISE_MULTIPLIER;
106
107 // use a sine wave to create a basic single tone
108 // from https://discussions.unity.com/t/generating-a-simple-sinewave/665023/16
109 case ProceduralType.SineWave:
110 float lastPhase = phase + (2 * Mathf.PI * (waveFrequency / SampleRate));
111
112 phase = lastPhase;
113 if (phase > 2 * Mathf.PI) // ensure that phase doesn't get exponentially larger
114 phase -= 2 * Mathf.PI;
115
116 return Mathf.Sin(lastPhase) * NOISE_MULTIPLIER;
117
118 // just silence in case you need it
119 case ProceduralType.Silence:
120 return 0;
121
122 default:
123 Debug.LogError("Attempting to get a sample from a procedural RadioTrack with an invalid ProceduralType- this should not be possible.");
124 return 0;
125 }
126 }
