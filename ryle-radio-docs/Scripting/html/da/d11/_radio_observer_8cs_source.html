95 private void Update()
96 {
97 // time to cache the update queue
98 // we need to do this as toDoOnUpdate could be changed on a different thread while we proceed through it, causing errors
99 List<Action> locUpdate;
100
101 // lock the update queue to clone it uninterrupted
102 lock (toDoOnUpdate)
103 {
104 locUpdate = new(toDoOnUpdate);
105 }
106
107 // move through the update queue to call the associated events
108 foreach (Action a in locUpdate)
109 {
110 a();
111
112 // remove this event from the update queue
113 lock (toDoOnUpdate)
114 toDoOnUpdate.Remove(a);
115 }

118 // link up each event to the associated track player
119 public void AssignEvents(RadioTrackPlayer _player)
120 {
121 foreach (ObserverEvent e in events)
122 {
123 switch (e.type)
124 {
125 // the _player events are mostly very similar- they give us the player it's called on, and the value
126 // we don't do anything with the player info in this script, but it's there for custom behaviour
127 //
128 // each sample, we check if a copy of this event has been called yet. If it has, we don't add another one, as it would hugely slow down at runtime
129 case EventType.OutputVolume:
130 if (hasEventBeenAdded[(int)EventType.OutputVolume]) break;
131 else hasEventBeenAdded[(int)EventType.OutputVolume] = true;
132
133 _player.OnVolume += (player, volume) => { StayEvent(e, volume); };
134 break;
135
136 case EventType.Gain:
137 if (hasEventBeenAdded[(int)EventType.Gain]) break;
138 else hasEventBeenAdded[(int)EventType.Gain] = true;
139
140 _player.OnGain -= (player, gain) => { StayEvent(e, gain); };
141 break;
142
143 case EventType.TunePower:
144 if (hasEventBeenAdded[(int)EventType.TunePower]) break;
145 else hasEventBeenAdded[(int)EventType.TunePower] = true;
146
147 _player.OnTunePower += (player, tunePower) => { StayEvent(e, tunePower); };
148 break;
149
150 case EventType.BroadcastPower:
151 if (hasEventBeenAdded[(int)EventType.BroadcastPower]) break;
152 else hasEventBeenAdded[(int)EventType.BroadcastPower] = true;
153
154 _player.OnBroadcastPower += (player, power) => { StayEvent(e, power); };
155 break;
156
157 case EventType.Insulation:
158 if (hasEventBeenAdded[(int)EventType.Insulation]) break;
159 else hasEventBeenAdded[(int)EventType.Insulation] = true;
160
161 _player.OnInsulation += (player, insulation) => { StayEvent(e, insulation); };
162 break;
163
164 // this is a single event rather than a value- there is no value associated with the track starting or ending, it just starts or ends
165 case EventType.TrackStarts:
166 _player.OnPlay += (player) => { TriggerEvent(e); };
167 break;
168
169 case EventType.TrackEnds:
170 _player.OnEnd += (player) => { TriggerEvent(e); };
171 break;
172
173 // this event watches the tune of the output- it doesn't actually use a player here
174 case EventType.OutputTune:
175 if (hasEventBeenAdded[(int)EventType.OutputTune]) break;
176 else hasEventBeenAdded[(int)EventType.OutputTune] = true;
177
178 output.OnTune += (tune) => { StayEvent(e, tune); };
179 break;
180
181 // an empty event gets no behaviour - gandhi
182 default:
183 case EventType.None:
184 break;
185 }
186 }

190 // a generic approach to an event tracking a value- we use a method here since we'd just be duplicating the code otherwise
191 private void StayEvent(ObserverEvent _event, float _value)
192 {
193 // if this event doesn't use a comparison, this method should not have been called- tell the user
194 if (!_event.NeedComparison)
195 {
196 Debug.LogWarning($"Attempting to use the FloatEvent method on trigger event {_event}! To fix, either modify the NeedComparison property in ObserverEvent, or change the method used in AssignEvents to TriggerEvent.");
197 return;
198 }
199
200 // evaluate the comparison
201 if (_event.EvaluateComparison(_value))
202 {
203 // if it passes,
204
205 // and this is the first time the event is called,
206 if (!_event.staying)
207 {
208 _event.staying = true;
209
210 // call the onTrigger event
211 lock (toDoOnUpdate)
212 {
213 toDoOnUpdate.Add(() => _event.onTrigger.Invoke(_value));
214 }
215 }
216 // and this is not the first time the event is called,
217 else
218 {
219 // call the onStay event
220 lock (toDoOnUpdate)
221 {
222 toDoOnUpdate.Add(() => _event.onStay.Invoke(_value));
223 }
224 }
225 }
226 else
227 {
228 // if it doesn't pass,
229
230 // and the event has just been called,
231 if (_event.staying)
232 {
233 _event.staying = false;
234
235 // call the onEnd event
236 lock (toDoOnUpdate)
237 {
238 toDoOnUpdate.Add(() => _event.onEnd.Invoke(_value));
239 }
240 }
241 }
242 }
243
244 // a generic approach to tracking a trigger rather than a value
245 private void TriggerEvent(ObserverEvent _event)
246 {
247 // if the event is triggered, call the onTrigger event- the other events are not applicable
248 lock (toDoOnUpdate)
249 toDoOnUpdate.Add(() => _event.onTrigger.Invoke(1));
250 }
251 }
