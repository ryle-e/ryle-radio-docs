23 /// The event an observer is looking for. For all options except TrackEnds and TrackStarts, we need an other value(s) to detect when this event occurs- for example, checking if volume is above a threshold

25 public enum EventType
26 {
27 OutputVolume, ///< The volume of the track: tune power * broadcast power * insulation
28 Gain, ///< The gain of the track: this is currently defined exclusively in the track's gain variable
29 TunePower, ///< The tune power of the track: how close the RadioOutput.Tune value is to the range of the track
30 BroadcastPower, ///< The broadcast power of the track: how close to any active RadioBroadcasters the output's transform is
31 Insulation, ///< The insulation of the track: the higher the value the less insulation- the power of any RadioInsulator the output is inside of
32 TrackEnds, ///< The track ends, or loops- this is a singular event
33 TrackStarts, ///< The track starts, or loops (happens after TrackEnds)- this is a singular event
34 OutputTune, ///< The tune on the RadioOutput is changed

117 private void Update()
118 {
119 // time to cache the update queue
120 // we need to do this as toDoOnUpdate could be changed on a different thread while we proceed through it, causing errors
121 List<Action> locUpdate;
122
123 // lock the update queue to clone it uninterrupted
124 lock (toDoOnUpdate)
125 {
126 locUpdate = new(toDoOnUpdate);
127 }
128
129 // move through the update queue to call the associated events
130 foreach (Action a in locUpdate)
131 {
132 a();
133
134 // remove this event from the update queue
135 lock (toDoOnUpdate)
136 toDoOnUpdate.Remove(a);
137 }

140 // link up each event to the associated track player
141 public void AssignEvents(RadioTrackPlayer _player)
142 {
143 foreach (ObserverEvent e in events)
144 {
145 switch (e.type)
146 {
147 // the _player events are mostly very similar- they give us the player it's called on, and the value
148 // we don't do anything with the player info in this script, but it's there for custom behaviour
149 //
150 // each sample, we check if a copy of this event has been called yet. If it has, we don't add another one, as it would hugely slow down at runtime
151 case EventType.OutputVolume:
152 if (hasEventBeenAdded[(int)EventType.OutputVolume]) break;
153 else hasEventBeenAdded[(int)EventType.OutputVolume] = true;
154
155 _player.OnVolume += (player, volume) => { StayEvent(e, volume); };
156 break;
157
158 case EventType.Gain:
159 if (hasEventBeenAdded[(int)EventType.Gain]) break;
160 else hasEventBeenAdded[(int)EventType.Gain] = true;
161
162 _player.OnGain -= (player, gain) => { StayEvent(e, gain); };
163 break;
164
165 case EventType.TunePower:
166 if (hasEventBeenAdded[(int)EventType.TunePower]) break;
167 else hasEventBeenAdded[(int)EventType.TunePower] = true;
168
169 _player.OnTunePower += (player, tunePower) => { StayEvent(e, tunePower); };
170 break;
171
172 case EventType.BroadcastPower:
173 if (hasEventBeenAdded[(int)EventType.BroadcastPower]) break;
174 else hasEventBeenAdded[(int)EventType.BroadcastPower] = true;
175
176 _player.OnBroadcastPower += (player, power) => { StayEvent(e, power); };
177 break;
178
179 case EventType.Insulation:
180 if (hasEventBeenAdded[(int)EventType.Insulation]) break;
181 else hasEventBeenAdded[(int)EventType.Insulation] = true;
182
183 _player.OnInsulation += (player, insulation) => { StayEvent(e, insulation); };
184 break;
185
186 // this is a single event rather than a value- there is no value associated with the track starting or ending, it just starts or ends
187 case EventType.TrackStarts:
188 _player.OnPlay += (player) => { TriggerEvent(e); };
189 break;
190
191 case EventType.TrackEnds:
192 _player.OnEnd += (player) => { TriggerEvent(e); };
193 break;
194
195 // this event watches the tune of the output- it doesn't actually use a player here
196 case EventType.OutputTune:
197 if (hasEventBeenAdded[(int)EventType.OutputTune]) break;
198 else hasEventBeenAdded[(int)EventType.OutputTune] = true;
199
200 output.OnTune += (tune) => { StayEvent(e, tune); };
201 break;
202
203 // an empty event gets no behaviour - gandhi
204 default:
205 case EventType.None:
206 break;
207 }
208 }

212 // a generic approach to an event tracking a value- we use a method here since we'd just be duplicating the code otherwise
213 private void StayEvent(ObserverEvent _event, float _value)
214 {
215 // if this event doesn't use a comparison, this method should not have been called- tell the user
216 if (!_event.NeedComparison)
217 {
218 Debug.LogWarning($"Attempting to use the FloatEvent method on trigger event {_event}! To fix, either modify the NeedComparison property in ObserverEvent, or change the method used in AssignEvents to TriggerEvent.");
219 return;
220 }
221
222 // evaluate the comparison
223 if (_event.EvaluateComparison(_value))
224 {
225 // if it passes,
226
227 // and this is the first time the event is called,
228 if (!_event.staying)
229 {
230 _event.staying = true;
231
232 // call the onTrigger event
233 lock (toDoOnUpdate)
234 {
235 toDoOnUpdate.Add(() => _event.onTrigger.Invoke(_value));
236 }
237 }
238 // and this is not the first time the event is called,
239 else
240 {
241 // call the onStay event
242 lock (toDoOnUpdate)
243 {
244 toDoOnUpdate.Add(() => _event.onStay.Invoke(_value));
245 }
246 }
247 }
248 else
249 {
250 // if it doesn't pass,
251
252 // and the event has just been called,
253 if (_event.staying)
254 {
255 _event.staying = false;
256
257 // call the onEnd event
258 lock (toDoOnUpdate)
259 {
260 toDoOnUpdate.Add(() => _event.onEnd.Invoke(_value));
261 }
262 }
263 }
264 }
265
266 // a generic approach to tracking a trigger rather than a value
267 private void TriggerEvent(ObserverEvent _event)
268 {
269 // if the event is triggered, call the onTrigger event- the other events are not applicable
270 lock (toDoOnUpdate)
271 toDoOnUpdate.Add(() => _event.onTrigger.Invoke(1));
272 }
273 }
