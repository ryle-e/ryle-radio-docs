245 // calculate the volume of the track at a specific tune
246 public float GetTunePower(float _tune, float _otherVolume)
247 {
248 if (_tune < range.x || _tune > range.y) // if the tune is out of this track's range, it cannot be heard
249 return 0;
250
251 float tunePower = rangeCurve.Evaluate(_tune.Remap(range.x, range.y, 0f, 1f)); // get the volume based on the tune and where it sits on the gain curve
252 float attenPower = 1f - (Mathf.Clamp01(_otherVolume) * attenuation); // get the volume based on attenuation and other playing trackWs
253
254 return tunePower * attenPower; // combine the values into one singular volume
255 }
