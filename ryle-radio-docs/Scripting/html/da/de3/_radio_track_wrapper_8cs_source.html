187 // limits the decimal points on the tune range, like an actual radio
188 private void ScaleRange()
189 {
190 range = new(
191 ((int)(range.x * RANGE_DECIMAL_MULTIPLIER)) / RANGE_DECIMAL_MULTIPLIER, // increase the size of the clampedRange clampedValue, then cut its decimals and shrink it back down
192 ((int)(range.y * RANGE_DECIMAL_MULTIPLIER)) / RANGE_DECIMAL_MULTIPLIER
193 );
194 }

197 // calculate the volume of the track at a specific tune
198 public float GetTunePower(float _tune, float _otherVolume)
199 {
200 if (_tune < range.x || _tune > range.y) // if the tune is out of this track's range, it cannot be heard
201 return 0;
202
203 float tunePower = rangeCurve.Evaluate(_tune.Remap(range.x, range.y, 0f, 1f)); // get the volume based on the tune and where it sits on the gain curve
204 float attenPower = 1f - (Mathf.Clamp01(_otherVolume) * attenuation); // get the volume based on attenuation and other playing trackWs
205
206 return tunePower * attenPower; // combine the values into one singular volume
207 }
