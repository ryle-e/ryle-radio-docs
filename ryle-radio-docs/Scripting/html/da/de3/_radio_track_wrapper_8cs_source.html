274 public static IRadioTrack CreateTrackEditor(string _name)
275 {
276 // get the index of the chosen track type
277 int index = Array.IndexOf(TrackNames, _name);
278
279 // if you somehow have an invalid track type, don't create anything

288 // return the new track

315 private void ScaleRange()
316 {
317 range = new(
318 ((int)(range.x * RANGE_DECIMAL_MULTIPLIER)) / RANGE_DECIMAL_MULTIPLIER, // increase the size of the clampedRange clampedValue, then cut its decimals and shrink it back down

330 public float GetTunePower(float _tune, float _otherVolume)
331 {
332 if (_tune < range.x || _tune > range.y) // if the tune is out of this track's range, it cannot be heard
333 return 0;
334
335 float tunePower = rangeCurve.Evaluate(_tune.Remap(range.x, range.y, 0f, 1f)); // get the volume based on the tune and where it sits on the gain curve
336 float attenPower = 1f - (Mathf.Clamp01(_otherVolume) * attenuation); // get the volume based on attenuation and other playing trackWs
337
338 return tunePower * attenPower; // combine the values into one singular volume
339 }
