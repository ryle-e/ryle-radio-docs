34 public void ReadClipAndForceToMono()
35 {
36 float[] allSamples = new float[clip.samples * clip.channels]; // all samples from both channels
37 Samples = new float[clip.samples]; // samples combined to one channel
38
39 // if the clip is invalid for some reason, tell the user
40 if (!clip.GetData(allSamples, 0))
41 {
42 Debug.LogError("Cannot access clip data from track " + clip.name);
43 return;
44 }
45
46 // for each sample (all channels)
47 for (int sample = 0; sample < clip.samples; sample++)
48 {
49 float combined = 0;
50
51 // for each channel in this sample, combine the audio
52 for (int channel = 0; channel < clip.channels; channel++)
53 combined += allSamples[(sample * clip.channels) + channel];
54
55 combined /= clip.channels; // find the average of the channels' audio
56 Samples[sample] = combined; // make the mono sample the average
57 }
