8 // a track that serves a multi-select- that is, a track that contains other tracks
9 // this is meant to emulate a radio station by storing many individual clips or audio then switching between them
10 // it can be used for really any purpose that requires switching tracks (at the end of the previous) though
11 // e.g ''''procedural'''' music by shuffling procedural sine waves
12
13 // this has a custom editor in StationRadioTrackEditor.cs
14 [System.Serializable]
15 public class StationRadioTrack : RadioTrack
16 {
17 public const string DISPLAY_NAME = "Station aka Multi-select";
18
19 // whether or not this station plays in a random or semi-random order
20 public bool randomSequence = true;
21
22 // how many other tracks need to be played before the same one is chosen again in a random sequence
23 // effectively stops the same track from playing back-to-back, and forces more variety in which tracks are played
24 // the number of tracks to be played before one can be repeated is round_down( (track_count - 1) * threshold )
25 // i.e with four tracks and a threshold of 0.5f, rounddown((4 - 1 == 3) * 0.5) == 1 other track will need to be played before a repeat
26 // i.e with four tracks and a threshold of 0.7f, rounddown((4 - 1 == 3) * 0.7) == 2 other tracks will need to be played before a repeat
27 // i.e with four tracks and a threshold of 1f, rounddown((4 - 1 == 3) * 1) == 3, aka all other tracks will need to be played before a repeat
28 // i.e with eleven tracks and a threshold of 0.8f, rounddown((11 - 1 == 10) * 0.8f == 8 other tracks will need to be played before a repeat
29 // do note that if this is set to 1, the tracks are forced to play in the same randomized sequence repeatedly
30 public float thresholdBeforeRepeats;
31
32 // the child tracks of this station
33 public List<StationRadioTrackWrapper> stationTrackWs;
34
35 // the currently playing child track
36 private int currentTrackIndex;
37
38 // audio is on a separate threat to UnityEngine.Random so we need to use System.Random instead
39 private System.Random random;
40
41 // this is stored as the number of plays that need to happen before this track can be played again.
42 // i.e if tracks A, B and C are being randomly chosen with a thresholdBeforeRepeats of 0.5f, they each need to have 1 other track play before it can repeat (see comments above thresholdBeforeRepeats)
43 // so if track A was just played after B and C, this array would look like [1, 0, -1]
44 // that is, track A needs another track to be played once before it can be repeated, B doesn't need any other tracks to play and thus can be repeated, and same for C
45 // (a number below 0 is treated as 0 for this system)
46 // see NextTrack() comments for more info
47 private int[] remainingTracksBeforeRepeat;
48
49 // if this track has no stations, we want to print an error- but only once, otherwise it freezes the editor
50 private bool hasPrintedError;
51
52 // the currently playing track
53 private StationRadioTrackWrapper CurrentTrackW => stationTrackWs[currentTrackIndex];
54

90 // selects the next track to play from the station
91 private void NextTrack()
92 {
93 if (stationTrackWs.Count == 0)
94 {
95 if (!hasPrintedError)
96 Debug.LogWarning($"Cannot play a StationRadioTrack, as there are no stations!");
97
98 hasPrintedError = true;
99 return;
100 }
101
102 // if the next track is randomly chosen
103 if (randomSequence)
104 {
105 // make a list for all of the tracks that can be chosen this time
106 List<int> selectFrom = new();
107
108 // find all tracks that can be played right now, and store their index from stationTrackWs
109 for (int i = 0; i < remainingTracksBeforeRepeat.Length; i++)
110 {
111 // if a track has <=0 tracks remaining before it can be played, then it can be played
112 if (remainingTracksBeforeRepeat[i] <= 0)
113 selectFrom.Add(i);
114 }
115
116 // select one track from the playable tracks
117 int trackIndex = selectFrom[random.Next(0, selectFrom.Count)];
118
119 // if each track needs to wait for a certain amount of other plays before it can be played again,
120 if (thresholdBeforeRepeats > 0)
121 {
122 // decrement the amount of plays each track needs before it can be played again
123 for (int t = 0; t < remainingTracksBeforeRepeat.Length; t++)
124 remainingTracksBeforeRepeat[t]--;
125
126 // set the selected track's amount to the maximum number (threshold * track count)
127 remainingTracksBeforeRepeat[trackIndex] = (int)((stationTrackWs.Count - 1) * thresholdBeforeRepeats);
128 }
129
130 // assign the chosen track
131 currentTrackIndex = trackIndex;
132 }
133 // otherwise if the station uses a set sequence
134 else
135 {
136 // increment the index
137 currentTrackIndex++;
138
139 // if the end of the station is reached, go back to the first track
140 if (currentTrackIndex >= stationTrackWs.Count)
141 currentTrackIndex = 0;
142 }
143
144 // assign the sample data based on the newly chosen track
145 SampleRate = CurrentTrackW.SampleRate;
146 SampleCount = CurrentTrackW.SampleCount;
147 }
148 }
